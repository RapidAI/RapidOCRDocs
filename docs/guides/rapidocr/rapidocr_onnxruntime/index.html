  2. pythonи„ҡжң¬дҪҝз”Ё

    • вҡ пёҸжіЁж„ҸпјҡеҲқе§ӢеҢ–RapidOCRеҸҜдёҚжҸҗдҫӣconfig.yamlпјҢй»ҳи®ӨдҪҝз”Ёе®үиЈ…зӣ®еҪ•дёӢзҡ„config.yamlгҖӮеҰӮжңүиҮӘе®ҡд№үйңҖжұӮпјҡ

      • дёҖжҳҜеҸҜзӣҙжҺҘйҖҡиҝҮеҲқе§ӢеҢ–еҸӮж•°дј е…ҘгҖӮиҜҰз»ҶеҸӮж•°еҸӮиҖғдёӢйқўе‘Ҫд»ӨиЎҢйғЁеҲҶпјҢе’Ңconfig.yamlеҹәжң¬еҜ№еә”гҖӮ
      • дәҢжҳҜеӨҚеҲ¶config.yamlпјҢиҮӘиЎҢжӣҙж”№пјҢз„¶еҗҺеҲқе§ӢеҢ–з»ҷеҮәгҖӮe.g. engine = RapidOCR(config_path="custom.yaml")

    • иҫ“е…ҘпјҡUnion[str, np.ndarray, bytes, Path]

    • иҫ“еҮәпјҡ

      • жңүеҖјпјҡ([[ж–Үжң¬жЎҶеқҗж Ү], ж–Үжң¬еҶ…е®№, зҪ®дҝЎеәҰ], жҺЁзҗҶж—¶й—ҙ)пјҢe.g. [[е·ҰдёҠ, еҸідёҠ, еҸідёӢ, е·ҰдёӢ], 'е°ҸжҳҺ', '0.99'], [0.02, 0.02, 0.85]
      • дёәз©әпјҡ(None, None)

    • зӨәдҫӢпјҡ

        import cv2
from rapidocr_onnxruntime import RapidOCR
# from rapidocr_openvino import RapidOCR

# RapidOCRеҸҜдј е…ҘеҸӮж•°еҸӮиҖғдёӢйқўзҡ„е‘Ҫд»ӨиЎҢйғЁеҲҶ
rapid_ocr = RapidOCR()

img_path = 'tests/test_files/ch_en_num.jpg'

# иҫ“е…Ҙж јејҸдёҖпјҡstr
result, elapse = rapid_ocr(img_path)

# иҫ“е…Ҙж јејҸдәҢпјҡnp.ndarray
img = cv2.imread('tests/test_files/ch_en_num.jpg')
result, elapse = rapid_ocr(img)

# иҫ“е…Ҙж јејҸдёүпјҡbytes
with open(img_path, 'rb') as f:
    img = f.read()
result, elapse = rapid_ocr(img)

# иҫ“е…Ҙж јејҸеӣӣпјҡPath
result, elapse = rapid_ocr(Path(img_path))
print(result)
